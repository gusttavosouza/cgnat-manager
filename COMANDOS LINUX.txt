Retorna o Nome da Interface
ifconfig | grep --text -i ': flags' | cut -f1 -d ':' | awk '{print $0}'

Pegar os IPs das interfaces
ifconfig | grep --text -i 'inet ' | cut -f1 -d '' | awk '{print $2}'

Pegar as Mascaras
ifconfig | grep --text -i 'inet ' | cut -f1 -d '' | awk '{print $4}'

Pegar pacotes entrando e saindo
ifconfig | grep --text -i 'RX packets ' | cut -f1 -d ':' | awk '{print $3}'
ifconfig | grep --text -i 'TX packets ' | cut -f1 -d ':' | awk '{print $3}'

Pegar Erros no TX e RX
ifconfig | grep --text -i 'RX errors' | cut -f1 -d '' | awk '{print $3}'
ifconfig | grep --text -i 'TX errors' | cut -f1 -d '' | awk '{print $3}'
