Pegar os IPs das interfaces
ip add | grep --text -i 'inet ' | cut -f1 -d ':' | awk '{print $2}'

Pegar pacotes entrando e saindo
ifconfig | grep --text -i 'RX packets ' | cut -f1 -d ':' | awk '{print $3}'
ifconfig | grep --text -i 'TX packets ' | cut -f1 -d ':' | awk '{print $3}'

Pegar trafego entrando e saindo
ifconfig | grep --text -i 'RX packets ' | cut -f1 -d ':' | awk '{print $5}'  //BYTES
ifconfig | grep --text -i 'TX packets ' | cut -f1 -d ':' | awk '{print $5}'  //BYTES
